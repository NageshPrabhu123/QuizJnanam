-- Remove the global unique constraint
ALTER TABLE ef_users DROP CONSTRAINT IF EXISTS ef_users_roll_no_unique;

-- Drop old indexes if they exist
DROP INDEX IF EXISTS uidx_rollno_year1_section;
DROP INDEX IF EXISTS uidx_rollno_year234_course;

-- 1st year: roll_no unique within same section only
-- e.g. two students in Section A cannot both have C01
-- but Section A and Section B can both have C01
CREATE UNIQUE INDEX uidx_rollno_1styear
ON ef_users (LOWER(roll_no), LOWER(section))
WHERE role = 'student' 
  AND year = 1 
  AND roll_no IS NOT NULL 
  AND roll_no != '';

-- 2nd+ year: roll_no unique within same year + course/branch only
-- e.g. two CSE 2nd year students cannot both have C01
-- but CSE and ISE can both have C01
CREATE UNIQUE INDEX uidx_rollno_2ndplusyear
ON ef_users (LOWER(roll_no), year, LOWER(course))
WHERE role = 'student' 
  AND year > 1 
  AND roll_no IS NOT NULL 
  AND roll_no != '';
